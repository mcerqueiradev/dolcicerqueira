CREATE PROCEDURE [dbo].[GetIngredientByRecipeId]
    @RecipeId INT
AS
BEGIN

SELECT L_Ingredients.*, Ingredient.Name, Measures.Unit  FROM L_Ingredients
inner join Ingredient on L_Ingredients.IngredientId = Ingredient.Id
inner join Measures on L_Ingredients.MeasureId = Measures.Id
WHERE RecipeId = @RecipeId

END
